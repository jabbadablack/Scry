cmake_minimum_required(VERSION 3.19)
project(SCRY C CXX)

if(NOT CMAKE_BUILD_TYPE AND NOT CMAKE_CONFIGURATION_TYPES)
    set(CMAKE_BUILD_TYPE Release CACHE STRING "Choose the type of build." FORCE)
endif()

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include(FetchContent)

# Fetch Core Dependencies
FetchContent_Declare(glfw GIT_REPOSITORY https://github.com/glfw/glfw.git GIT_TAG master)
FetchContent_Declare(taskflow GIT_REPOSITORY https://github.com/taskflow/taskflow.git GIT_TAG master)
FetchContent_Declare(doctest GIT_REPOSITORY https://github.com/doctest/doctest.git GIT_TAG v2.4.11)
FetchContent_Declare(eigen GIT_REPOSITORY https://gitlab.com/libeigen/eigen.git GIT_TAG 3.4.0)
FetchContent_Declare(entt GIT_REPOSITORY https://github.com/skypjack/entt.git GIT_TAG v3.13.0)
FetchContent_Declare(stb GIT_REPOSITORY https://github.com/nothings/stb.git GIT_TAG master)
FetchContent_Declare(assimp GIT_REPOSITORY https://github.com/assimp/assimp.git GIT_TAG v5.3.1)

# Fetch ImGui & Tracy into Core
FetchContent_Declare(imgui GIT_REPOSITORY https://github.com/ocornut/imgui.git GIT_TAG docking)
FetchContent_Declare(tracy GIT_REPOSITORY https://github.com/wolfpld/tracy.git GIT_TAG master)

# Dependency Configs
set(ASSIMP_BUILD_ASSIMP_TOOLS OFF CACHE BOOL "" FORCE)
set(ASSIMP_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(GLFW_BUILD_DOCS OFF CACHE BOOL "" FORCE)

# --- TRACY POTATO PC CONSTRAINTS ---
set(TRACY_ENABLE ON CACHE BOOL "" FORCE)
set(TRACY_ON_DEMAND ON CACHE BOOL "" FORCE)        # Only allocate heavy queues when the server actually connects
set(TRACY_NO_SYSTEM_TRACING ON CACHE BOOL "" FORCE)# Disable heavy OS-level context switch tracking
set(TRACY_NO_FRAME_IMAGE ON CACHE BOOL "" FORCE)   # Prevent capturing framebuffer screenshots to RAM
set(TRACY_NO_CALLSTACK ON CACHE BOOL "" FORCE)     # Disable automatic callstack unwinding (massive RAM saver)

FetchContent_MakeAvailable(glfw taskflow doctest entt assimp tracy)

# ImGui setup
FetchContent_GetProperties(imgui)
if(NOT imgui_POPULATED)
    FetchContent_Populate(imgui)
endif()
file(WRITE ${imgui_SOURCE_DIR}/CMakeLists.txt "cmake_minimum_required(VERSION 3.19)\nproject(imgui CXX)\nadd_library(imgui STATIC imgui.cpp imgui_demo.cpp imgui_draw.cpp imgui_tables.cpp imgui_widgets.cpp backends/imgui_impl_glfw.cpp)\ntarget_include_directories(imgui PUBLIC \${CMAKE_CURRENT_SOURCE_DIR} \${CMAKE_CURRENT_SOURCE_DIR}/backends)\ntarget_link_libraries(imgui PUBLIC glfw)\nadd_library(imgui::imgui ALIAS imgui)\n")
add_subdirectory(${imgui_SOURCE_DIR} ${imgui_BINARY_DIR})

# Header-only setups
FetchContent_GetProperties(eigen)
if(NOT eigen_POPULATED)
    FetchContent_Populate(eigen)
endif()
FetchContent_GetProperties(stb)
if(NOT stb_POPULATED)
    FetchContent_Populate(stb)
endif()

add_library(${PROJECT_NAME} INTERFACE)
add_library(${PROJECT_NAME}::${PROJECT_NAME} ALIAS ${PROJECT_NAME})

target_include_directories(${PROJECT_NAME} INTERFACE
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<BUILD_INTERFACE:${eigen_SOURCE_DIR}>
    $<BUILD_INTERFACE:${stb_SOURCE_DIR}>
)

target_link_libraries(${PROJECT_NAME} INTERFACE glfw Taskflow EnTT::EnTT assimp imgui::imgui TracyClient)
if(WIN32)
    target_link_libraries(${PROJECT_NAME} INTERFACE winmm)
endif()

# POTATO PC / EXTREME RAM REDUCTION
if(MSVC)
    target_compile_options(${PROJECT_NAME} INTERFACE $<$<NOT:$<CONFIG:Debug>>:/GR-> /Gy /Gw)
    target_link_options(${PROJECT_NAME} INTERFACE /OPT:REF /OPT:ICF)
else()
    target_compile_options(${PROJECT_NAME} INTERFACE $<$<NOT:$<CONFIG:Debug>>:-fno-rtti> -ffunction-sections -fdata-sections)
    target_link_options(${PROJECT_NAME} INTERFACE -Wl,--gc-sections)
endif()

target_compile_definitions(${PROJECT_NAME} INTERFACE ENTT_PACKED_PAGE=128 ENTT_SPARSE_PAGE=128)

file(GLOB_RECURSE SCRY_HEADERS CONFIGURE_DEPENDS "include/*.h" "include/*.hpp" "include/*.inl")
target_sources(${PROJECT_NAME} INTERFACE FILE_SET HEADERS BASE_DIRS include FILES ${SCRY_HEADERS})

add_subdirectory(modules/renderer)
add_subdirectory(tests)
add_subdirectory(samples/engine-basic)

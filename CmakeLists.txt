    add_subdirectory(tests)

    # Exclude tests from ALL builds by default
    set_target_properties(scry_tests PROPERTIES EXCLUDE_FROM_ALL TRUE)

    # Re-enable for Debug in Multi-Config generators
    if(CMAKE_CONFIGURATION_TYPES)
        foreach(cfg ${CMAKE_CONFIGURATION_TYPES})
            if(cfg STREQUAL "Debug")
                string(TOUPPER ${cfg} cfg_upper)
                set_target_properties(scry_tests PROPERTIES EXCLUDE_FROM_ALL_${cfg_upper} FALSE)
            endif()
        endforeach()
    endif()

    add_custom_target(run_tests
        COMMAND $<TARGET_FILE:scry_tests> --reporter compact
        DEPENDS scry_tests
        WORKING_DIRECTORY "$<TARGET_FILE_DIR:scry_tests>"
        COMMENT "Running Scry test suite"
        VERBATIM
    )

    set_target_properties(run_tests PROPERTIES EXCLUDE_FROM_ALL TRUE)
    if(CMAKE_CONFIGURATION_TYPES)
        foreach(cfg ${CMAKE_CONFIGURATION_TYPES})
            if(cfg STREQUAL "Debug")
                string(TOUPPER ${cfg} cfg_upper)
                set_target_properties(run_tests PROPERTIES EXCLUDE_FROM_ALL_${cfg_upper} FALSE)
            endif()
        endforeach()
    endif()

# ─── Shader Compilation ───
if(TARGET shaderc)
    set(SHADERC_EXE shaderc)
else()
    set(SHADERC_EXE ${CMAKE_RUNTIME_OUTPUT_DIRECTORY}/shaderc)
endif()

function(compile_bgfx_shader SHADER_FILE TYPE PLATFORM PROFILE OUTPUT_FILE)
    add_custom_command(
        OUTPUT ${OUTPUT_FILE}
        COMMAND ${SHADERC_EXE} -f ${SHADER_FILE} -o ${OUTPUT_FILE} --type ${TYPE} --platform ${PLATFORM} --profile ${PROFILE} -i "${bgfx_cmake_SOURCE_DIR}/bgfx/src"
        DEPENDS ${SHADER_FILE} shaderc
        COMMENT "Compiling shader ${SHADER_FILE} -> ${OUTPUT_FILE}"
    )
endfunction()

if(MSVC)
    set(BGFX_PLATFORM "windows")
    set(BGFX_VS_PROFILE "vs_5_0")
    set(BGFX_FS_PROFILE "ps_5_0")
    set(BGFX_CS_PROFILE "cs_5_0")
elseif(APPLE)
    set(BGFX_PLATFORM "osx")
    set(BGFX_VS_PROFILE "metal")
    set(BGFX_FS_PROFILE "metal")
    set(BGFX_CS_PROFILE "metal")
else()
    set(BGFX_PLATFORM "linux")
    set(BGFX_VS_PROFILE "spirv")
    set(BGFX_FS_PROFILE "spirv")
    set(BGFX_CS_PROFILE "spirv")
endif()

set(SHADER_SRC_DIR "${CMAKE_CURRENT_SOURCE_DIR}/assets/raw/shaders")
set(SHADER_BIN_DIR "${CMAKE_BINARY_DIR}/bin/assets/cooked/shaders")

file(MAKE_DIRECTORY ${SHADER_BIN_DIR})

compile_bgfx_shader("${SHADER_SRC_DIR}/vs_mesh.sc"       "vertex"   ${BGFX_PLATFORM} ${BGFX_VS_PROFILE} "${SHADER_BIN_DIR}/vs_mesh.bin")
compile_bgfx_shader("${SHADER_SRC_DIR}/fs_mesh.sc"       "fragment" ${BGFX_PLATFORM} ${BGFX_FS_PROFILE} "${SHADER_BIN_DIR}/fs_mesh.bin")
compile_bgfx_shader("${SHADER_SRC_DIR}/cs_indirect.sc"   "compute"  ${BGFX_PLATFORM} ${BGFX_CS_PROFILE} "${SHADER_BIN_DIR}/cs_indirect.bin")

add_custom_target(cook_shaders DEPENDS
    "${SHADER_BIN_DIR}/vs_mesh.bin"
    "${SHADER_BIN_DIR}/fs_mesh.bin"
    "${SHADER_BIN_DIR}/cs_indirect.bin"
)

# ─── cook_assets custom target ───────────────────────────────────────────────
add_custom_target(cook_assets
    COMMAND $<TARGET_FILE:scry_cooker>
            "${CMAKE_CURRENT_SOURCE_DIR}/assets/raw"
            "${CMAKE_BINARY_DIR}/bin/assets/cooked"
    DEPENDS scry_cooker cook_shaders
    COMMENT "Cooking raw assets into cooked binary blobs"
    VERBATIM
)

add_dependencies(scry cook_assets)

# MinGW fix: DiligentCore's VK shared DLL linker uses --version-script=export.map
# with a RELATIVE path, but ninja runs the linker from ${CMAKE_BINARY_DIR}.
# DiligentCore writes export.map to its own build subdir; copy it to the root.
if(MINGW)
    set(_diligent_vk_export_map
        "${diligentcore_BINARY_DIR}/Graphics/GraphicsEngineVulkan/export.map")
    if(EXISTS "${_diligent_vk_export_map}")
        configure_file("${_diligent_vk_export_map}" "${CMAKE_BINARY_DIR}/export.map" COPYONLY)
    else()
        file(WRITE "${CMAKE_BINARY_DIR}/export.map"
            "{ global: *GetEngineFactoryVk*; local: *; };\n")
    endif()
endif()

endif()

# ─── Engine Library ──────────────────────────────────────────────────────────

file(GLOB_RECURSE SCRY_SRC_C   CONFIGURE_DEPENDS "src/*.c"   "src/systems/*.c")
file(GLOB_RECURSE SCRY_SRC_CPP CONFIGURE_DEPENDS "src/*.cpp" "src/renderer/*.cpp")
file(GLOB_RECURSE SCRY_INC     CONFIGURE_DEPENDS "include/*.h")

add_library(scry SHARED ${SCRY_SRC_C} ${SCRY_SRC_CPP} ${SCRY_INC})
target_compile_options(scry PRIVATE 
    $<$<COMPILE_LANGUAGE:C>:${SCRY_C_FLAGS}>
    $<$<COMPILE_LANGUAGE:CXX>:${SCRY_CXX_FLAGS}>
)

if(MSVC)
    target_compile_options(scry PRIVATE /O2 /fp:fast /arch:AVX2)
else()
    target_compile_options(scry PRIVATE -O3 -ffast-math -mavx2)
endif()

target_include_directories(scry PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>
    "${cglm_SOURCE_DIR}/include"
)

# DiligentCore root — SYSTEM suppresses -Wpedantic warnings from their headers
target_include_directories(scry SYSTEM PRIVATE
    "${diligentcore_SOURCE_DIR}"
)

if(WIN32)
    target_compile_definitions(scry PRIVATE PLATFORM_WIN32=1 ENGINE_EXPORT)
elseif(UNIX AND NOT APPLE)
    target_compile_definitions(scry PRIVATE PLATFORM_LINUX=1 ENGINE_EXPORT)
elseif(APPLE)
    target_compile_definitions(scry PRIVATE PLATFORM_MACOS=1 ENGINE_EXPORT)
endif()

target_link_libraries(scry
    PUBLIC
        flecs
        glfw
        yyjson
        cglm
    PRIVATE
        Diligent-GraphicsEngineVk-static
)

    set_target_properties(scry_tests PROPERTIES EXCLUDE_FROM_ALL TRUE)

    if(CMAKE_CONFIGURATION_TYPES)
        foreach(cfg ${CMAKE_CONFIGURATION_TYPES})
            if(cfg STREQUAL "Debug")
                string(TOUPPER ${cfg} cfg_upper)
                set_target_properties(scry_tests PROPERTIES EXCLUDE_FROM_ALL_${cfg_upper} FALSE)
            endif()
        endforeach()
    endif()

    add_custom_target(run_tests
        COMMAND $<TARGET_FILE:scry_tests> --reporter compact
        DEPENDS scry_tests
        WORKING_DIRECTORY "$<TARGET_FILE_DIR:scry_tests>"
        COMMENT "Running Scry test suite"
        VERBATIM
    )

    set_target_properties(run_tests PROPERTIES EXCLUDE_FROM_ALL TRUE)
    if(CMAKE_CONFIGURATION_TYPES)
        foreach(cfg ${CMAKE_CONFIGURATION_TYPES})
            if(cfg STREQUAL "Debug")

# ─── Shader copy ─────────────────────────────────────────────────────────────
# DiligentCore compiles HLSL at runtime; just copy mesh.hlsl to the output tree.

set(SHADER_DIR_SRC "${CMAKE_CURRENT_SOURCE_DIR}/assets/raw/shaders")
set(SHADER_DIR_DST "${CMAKE_BINARY_DIR}/bin/assets/raw/shaders")

file(MAKE_DIRECTORY "${SHADER_DIR_DST}")

add_custom_command(
    OUTPUT  "${SHADER_DIR_DST}/mesh.hlsl"
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "${SHADER_DIR_SRC}/mesh.hlsl" "${SHADER_DIR_DST}/mesh.hlsl"
    DEPENDS "${SHADER_DIR_SRC}/mesh.hlsl"
    COMMENT "Copying mesh.hlsl to output"
)

add_custom_command(
    OUTPUT  "${SHADER_DIR_DST}/cull.hlsl"
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "${SHADER_DIR_SRC}/cull.hlsl" "${SHADER_DIR_DST}/cull.hlsl"
    DEPENDS "${SHADER_DIR_SRC}/cull.hlsl"
    COMMENT "Copying cull.hlsl to output"
)

add_custom_target(copy_shaders
    DEPENDS
        "${SHADER_DIR_DST}/mesh.hlsl"
        "${SHADER_DIR_DST}/cull.hlsl"
)

# ─── cook_assets custom target ───────────────────────────────────────────────
add_custom_target(cook_assets
    COMMAND $<TARGET_FILE:scry_cooker>
            "${CMAKE_CURRENT_SOURCE_DIR}/assets/raw"
            "${CMAKE_BINARY_DIR}/bin/assets/cooked"
    DEPENDS scry_cooker copy_shaders
    COMMENT "Cooking raw assets into cooked binary blobs"
    VERBATIM
)

add_dependencies(scry cook_assets)

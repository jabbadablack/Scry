cmake_minimum_required(VERSION 3.19)
project(SCRY VERSION 1.0.0 LANGUAGES C CXX)


set(CMAKE_POLICY_VERSION_MINIMUM 3.5)


# Require C++20 and C11
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_SCAN_FOR_MODULES OFF)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Set global output directories for all targets
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)


# Configure mimalloc options
set(MI_OVERRIDE ON CACHE BOOL "Override standard malloc" FORCE)
set(MI_BUILD_SHARED OFF CACHE BOOL "Build mimalloc as a shared library" FORCE)
set(MI_BUILD_OBJECT OFF CACHE BOOL "Build mimalloc object files" FORCE)
set(MI_BUILD_TESTS OFF CACHE BOOL "Build mimalloc tests" FORCE)
set(MI_BUILD_STATIC ON CACHE BOOL "Build mimalloc as static library" FORCE)

# Configure SDL3 options to minimize compile time
set(SDL_SHARED OFF CACHE BOOL "Build shared SDL library" FORCE)
set(SDL_STATIC ON CACHE BOOL "Build static SDL library" FORCE)
set(SDL_TESTS OFF CACHE BOOL "Build SDL tests" FORCE)
set(SDL_INSTALL OFF CACHE BOOL "Install SDL" FORCE)
set(SDL_AUDIO OFF CACHE BOOL "Disable SDL Audio subsystem" FORCE)
set(SDL_JOYSTICK OFF CACHE BOOL "Disable SDL Joystick subsystem" FORCE)
set(SDL_HAPTIC OFF CACHE BOOL "Disable SDL Haptic subsystem" FORCE)
set(SDL_SENSOR OFF CACHE BOOL "Disable SDL Sensor subsystem" FORCE)
set(SDL_CAMERA OFF CACHE BOOL "Disable SDL Camera subsystem" FORCE)

# Configure Flecs options
set(FLECS_SHARED ON CACHE BOOL "Build Flecs shared library" FORCE)
set(FLECS_STATIC OFF CACHE BOOL "Build Flecs static library" FORCE)
set(FLECS_TESTS OFF CACHE BOOL "Build Flecs tests" FORCE)

# Configure Catch2 options
set(CATCH_INSTALL_HELPERS OFF CACHE BOOL "" FORCE)
set(CATCH_BUILD_TESTING OFF CACHE BOOL "" FORCE)

# Configure quill options
set(QUILL_ENABLE_TESTS OFF CACHE BOOL "" FORCE)
set(QUILL_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)

# Configure Eigen options
set(EIGEN_BUILD_DOC OFF CACHE BOOL "" FORCE)
set(EIGEN_BUILD_TESTING OFF CACHE BOOL "" FORCE)

# CMake Options
option(SCRY_BUILD_EXAMPLES "Build the example game project (examples/sandbox)" ON)
option(SCRY_BUILD_TESTS    "Build and auto-run the test suite after every build" ON)

    GIT_TAG v4.1.5
)

FetchContent_Declare(
    yyjson
    GIT_REPOSITORY https://github.com/ibireme/yyjson.git
    GIT_TAG 0.10.0
)

FetchContent_Declare(
    Catch2
    GIT_REPOSITORY https://github.com/catchorg/Catch2.git
    GIT_TAG v3.5.2
)

FetchContent_Declare(
    libassert
    GIT_REPOSITORY https://github.com/jeremy-rifkin/libassert.git

FetchContent_Declare(

FetchContent_Declare(

    eigen

FetchContent_MakeAvailable(mimalloc SDL3 flecs yyjson Catch2 libassert quill string_id)

# Populate and patch eigen manually to avoid target collision on "uninstall"
FetchContent_GetProperties(eigen)
if(NOT eigen_POPULATED)
    FetchContent_Populate(eigen)
    if(EXISTS "${eigen_SOURCE_DIR}/CMakeLists.txt")
        file(READ "${eigen_SOURCE_DIR}/CMakeLists.txt" EIGEN_CMAKE)
        string(REPLACE "add_custom_target ( uninstall" "add_custom_target ( eigen_uninstall" EIGEN_CMAKE "${EIGEN_CMAKE}")
        string(REPLACE "add_custom_target(uninstall" "add_custom_target(eigen_uninstall" EIGEN_CMAKE "${EIGEN_CMAKE}")
        file(WRITE "${eigen_SOURCE_DIR}/CMakeLists.txt" "${EIGEN_CMAKE}")
    endif()
    add_subdirectory(${eigen_SOURCE_DIR} ${eigen_BINARY_DIR})
endif()

# Define rigorous warning flags for Scry targets
if(CMAKE_CXX_COMPILER_ID MATCHES "GNU" OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
    set(SCRY_WARNING_FLAGS -Wall -Wextra -Wpedantic -Werror -Wno-tautological-compare -Wno-deprecated-enum-enum-conversion -Wno-deprecated-copy -Wno-deprecated-declarations -Wno-template-body)
elseif(MSVC)
    set(SCRY_WARNING_FLAGS /W4 /WX)
endif()

# Load plugin/game helper macros.  These are also exported to the build tree
# so that downstream projects can call scry_add_plugin() after find_package(Scry).
include(cmake/ScryHelpers.cmake)

# ─── Engine Library ──────────────────────────────────────────────────────────

file(GLOB_RECURSE SCRY_SRC CONFIGURE_DEPENDS "src/*.cpp")
file(GLOB_RECURSE SCRY_INC CONFIGURE_DEPENDS "include/*.hpp" "include/*.h")

add_library(scry SHARED ${SCRY_SRC} ${SCRY_INC})
target_compile_options(scry PRIVATE ${SCRY_WARNING_FLAGS})

target_include_directories(scry PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>
)

target_compile_definitions(scry PRIVATE SCRY_EXPORT)

target_link_libraries(scry
    PUBLIC
        libassert::assert
        eigen
        quill::quill
    PRIVATE
        mimalloc-static
        SDL3-static
        flecs
        yyjson
        foonathan_string_id
)
target_include_directories(scry PRIVATE
    "${mimalloc_SOURCE_DIR}/include"
    "${string_id_SOURCE_DIR}"
    "${string_id_BINARY_DIR}"
)

# Precompile Eigen/Dense once for the scry target — it's a very heavy header
target_precompile_headers(scry PRIVATE <Eigen/Dense>)

# Suppress warnings from third-party headers
foreach(dep eigen quill foonathan_string_id flecs yyjson)
    if(TARGET ${dep})
        set_target_properties(${dep} PROPERTIES
            INTERFACE_SYSTEM_INCLUDE_DIRECTORIES
                "$<TARGET_PROPERTY:${dep},INTERFACE_INCLUDE_DIRECTORIES>")
    endif()
endforeach()

# ─── Tests ───────────────────────────────────────────────────────────────────
# When SCRY_BUILD_TESTS is ON the test suite is built AND automatically executed
# as part of every cmake --build invocation (via the run_tests ALL target).

if(SCRY_BUILD_TESTS)
    add_subdirectory(tests)

    # Run the test binary directly rather than through CTest.
    # enable_testing() at root scope pulls in Eigen's 780+ test targets;
    # invoking scry_tests directly sidesteps that entirely.
    add_custom_target(run_tests ALL
        COMMAND $<TARGET_FILE:scry_tests> --reporter compact
        DEPENDS scry_tests
        WORKING_DIRECTORY "$<TARGET_FILE_DIR:scry_tests>"
        COMMENT "Running Scry test suite"
    )
endif()

# ─── Example game project ────────────────────────────────────────────────────
# The sandbox is its own isolated CMake project (has cmake_minimum_required +
# project()) — it just happens to be added here via add_subdirectory so that
# one cmake --build from the engine root builds everything together.
# Plugins are NOT declared here; the sandbox's own project.json drives which
# plugins are built, connecting them to the engine instance the sandbox uses.

if(SCRY_BUILD_EXAMPLES)
    set(SCRY_PLUGINS_DIR "${CMAKE_CURRENT_SOURCE_DIR}/plugins" CACHE PATH
        "Root directory that contains all Scry plugin sub-projects")
    add_subdirectory(examples/sandbox)
endif()

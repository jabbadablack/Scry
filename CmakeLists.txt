# GCC emits a false-positive -Warray-bounds in Assimp's X3DGeoHelper when
# vector::back() is inlined and the optimizer's range analysis goes wrong.
if(TARGET assimp AND CMAKE_CXX_COMPILER_ID STREQUAL "GNU")
    target_compile_options(assimp PRIVATE -Wno-array-bounds)
endif()

# --- Dead Code Elimination (DCE) & Optimization ---
# Forces the compiler to isolate functions/data, allowing the linker to garbage-collect unused code.
if(MSVC)
    # /Gy: Enable function-level linking (puts each function in its own COMDAT)
    # /Gw: Optimize global data (puts data in individual COMDATs)
    target_compile_options(${PROJECT_NAME} INTERFACE /Gy /Gw)
    
    # /OPT:REF: Eliminate unreferenced functions/data
    # /OPT:ICF: Fold identical COMDAT sections
    target_link_options(${PROJECT_NAME} INTERFACE /OPT:REF /OPT:ICF)
else()
    # -ffunction-sections & -fdata-sections: Isolate functions and data into distinct sections
    target_compile_options(${PROJECT_NAME} INTERFACE -ffunction-sections -fdata-sections)
    
    # -Wl,--gc-sections: Instruct the linker to aggressively drop unreferenced sections
    target_link_options(${PROJECT_NAME} INTERFACE -Wl,--gc-sections)
endif()

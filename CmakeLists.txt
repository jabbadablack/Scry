# Catch2

# SOURCE_DIR is set so the parent dir (_deps) resolves #include <DiligentCore/...>
FetchContent_Declare(DiligentCore
    GIT_REPOSITORY https://github.com/DiligentGraphics/DiligentCore.git
    GIT_TAG        master
    SOURCE_DIR     "${CMAKE_BINARY_DIR}/_deps/DiligentCore"
)

# DiligentCore is processed first because it internally fetches mimalloc via
# its own FetchContent_MakeAvailable call.  Processing ours first causes a
# duplicate add_library error in CMake < 3.24 (no double-add-subdirectory guard).
# By letting DiligentCore "own" mimalloc we avoid the conflict; the if(NOT TARGET)
# guard below makes sure we don't create a second mimalloc-static on our side.
FetchContent_MakeAvailable(assimp stb DiligentCore meshoptimizer)

if(NOT TARGET mimalloc-static)
    FetchContent_MakeAvailable(mimalloc)
endif()

FetchContent_MakeAvailable(SDL3 flecs yyjson Catch2 cpptrace libassert string_id enkiTS eigen)

endif()

include(cmake/EngineHelpers.cmake)

# ─── Engine Library ──────────────────────────────────────────────────────────

file(GLOB_RECURSE SCRY_SRC CONFIGURE_DEPENDS "src/*.cpp")
file(GLOB_RECURSE SCRY_INC CONFIGURE_DEPENDS "include/*.hpp" "include/*.h")

add_library(scry SHARED ${SCRY_SRC} ${SCRY_INC})
target_compile_options(scry PRIVATE ${SCRY_WARNING_FLAGS})

target_include_directories(scry PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>
    "${eigen_SOURCE_DIR}"
    "${string_id_SOURCE_DIR}"
)

# DiligentCore include paths:
#   _deps              → lets our headers use <DiligentCore/Graphics/...>
#   _deps/DiligentCore → lets DiligentCore's own headers find each other as <Primitives/...>
target_include_directories(scry PRIVATE
    "${CMAKE_BINARY_DIR}/_deps"
    "${CMAKE_BINARY_DIR}/_deps/DiligentCore"
)

# DiligentCore sets PLATFORM_WIN32/LINUX/MACOS via add_compile_definitions() which is a
# directory property and does NOT propagate through target_link_libraries to external targets.
if(WIN32)
    target_compile_definitions(scry PRIVATE PLATFORM_WIN32=1)
elseif(UNIX AND NOT APPLE)
    target_compile_definitions(scry PRIVATE PLATFORM_LINUX=1)
elseif(APPLE)
    target_compile_definitions(scry PRIVATE PLATFORM_MACOS=1)
endif()

# Backend selection: D3D12 for MSVC, Vulkan for GCC/MinGW
if(MSVC)
    target_compile_definitions(scry PRIVATE ENGINE_EXPORT SCRY_BACKEND_DX12)
    target_link_libraries(scry PRIVATE Diligent-GraphicsEngineDX12-static)
else()
    target_compile_definitions(scry PRIVATE ENGINE_EXPORT SCRY_BACKEND_VULKAN)
    target_link_libraries(scry PRIVATE Diligent-GraphicsEngineVk-static)
endif()

target_link_libraries(scry
    PUBLIC
        mimalloc-static
        flecs
        SDL3::SDL3-static
        yyjson
        libassert::assert
        foonathan_string_id
        enkiTS
)

# ─── Tests ───────────────────────────────────────────────────────────────────

if(SCRY_BUILD_TESTS)
    add_subdirectory(tests)

    # Exclude tests from ALL builds by default
    set_target_properties(scry_tests PROPERTIES EXCLUDE_FROM_ALL TRUE)
    
    # Re-enable for Debug in Multi-Config generators
    if(CMAKE_CONFIGURATION_TYPES)
        foreach(cfg ${CMAKE_CONFIGURATION_TYPES})
            if(cfg STREQUAL "Debug")
                string(TOUPPER ${cfg} cfg_upper)
                set_target_properties(scry_tests PROPERTIES EXCLUDE_FROM_ALL_${cfg_upper} FALSE)
            endif()
        endforeach()
    endif()

    add_custom_target(run_tests
        COMMAND $<TARGET_FILE:scry_tests> --reporter compact
        DEPENDS scry_tests
        WORKING_DIRECTORY "$<TARGET_FILE_DIR:scry_tests>"
        COMMENT "Running Scry test suite"
        VERBATIM
    )
    
    set_target_properties(run_tests PROPERTIES EXCLUDE_FROM_ALL TRUE)
    if(CMAKE_CONFIGURATION_TYPES)
        foreach(cfg ${CMAKE_CONFIGURATION_TYPES})
            if(cfg STREQUAL "Debug")
                string(TOUPPER ${cfg} cfg_upper)
                set_target_properties(run_tests PROPERTIES EXCLUDE_FROM_ALL_${cfg_upper} FALSE)
            endif()
        endforeach()
    endif()
endif()

# ─── Asset cooker tool ────────────────────────────────────────────────────────
# Standalone executable — links ONLY against assimp + stb.  No Scry runtime.

add_executable(scry_cooker tools/cooker.cpp)

target_compile_features(scry_cooker PRIVATE cxx_std_17)

target_include_directories(scry_cooker PRIVATE
    "${CMAKE_CURRENT_SOURCE_DIR}/include"   # for engine/CookedAsset.h
    "${stb_SOURCE_DIR}"                     # for stb_image.h
    "${assimp_SOURCE_DIR}/include"
    "${assimp_BINARY_DIR}/include"          # for generated assimp/config.h
    "${meshoptimizer_SOURCE_DIR}/src"
)

target_link_libraries(scry_cooker PRIVATE assimp::assimp meshoptimizer)

# ─── cook_assets custom target ───────────────────────────────────────────────
# Runs scry_cooker over assets/raw → bin/assets/cooked before scry is linked.

add_custom_target(cook_assets
    COMMAND $<TARGET_FILE:scry_cooker>
            "${CMAKE_CURRENT_SOURCE_DIR}/assets/raw"
            "${CMAKE_BINARY_DIR}/bin/assets/cooked"
    DEPENDS scry_cooker
    COMMENT "Cooking raw assets into cooked binary blobs"
    VERBATIM
)

add_dependencies(scry cook_assets)

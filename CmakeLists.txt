cmake_minimum_required(VERSION 3.19)
project(SCRY VERSION 1.0.0 LANGUAGES C CXX)

set(CMAKE_POLICY_VERSION_MINIMUM 3.5)

# Require C++17 and C11
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Configure output directories
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)

# Build options
option(SCRY_BUILD_TESTS "Build unit tests" ON)
option(SCRY_BUILD_EXAMPLES "Build example sandbox" ON)

include(FetchContent)

set(CMAKE_SKIP_INSTALL_RULES ON CACHE BOOL "" FORCE)
set(BUILD_TESTING OFF CACHE BOOL "" FORCE)

# ─── Dependencies (Local Source Overrides) ───

FetchContent_Declare(mimalloc SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/external/mimalloc")
set(MI_OVERRIDE ON CACHE BOOL "" FORCE)
set(MI_BUILD_SHARED OFF CACHE BOOL "" FORCE)
set(MI_BUILD_STATIC ON CACHE BOOL "" FORCE)

FetchContent_Declare(SDL3 SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/external/SDL3")
set(SDL_SHARED OFF CACHE BOOL "" FORCE)
set(SDL_STATIC ON CACHE BOOL "" FORCE)

# Build Flecs as SHARED to ensure global state is consistent across DLL boundaries
FetchContent_Declare(flecs SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/external/flecs")
set(FLECS_STATIC OFF CACHE BOOL "" FORCE)
set(FLECS_SHARED ON CACHE BOOL "" FORCE)

FetchContent_Declare(yyjson SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/external/yyjson")
FetchContent_Declare(Catch2 SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/external/Catch2")
FetchContent_Declare(cpptrace SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/external/cpptrace")
FetchContent_Declare(libassert SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/external/libassert")
FetchContent_Declare(string_id SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/external/string_id")
FetchContent_Declare(enkiTS SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/external/enkiTS")
FetchContent_Declare(eigen SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/external/eigen")

FetchContent_MakeAvailable(mimalloc SDL3 flecs yyjson Catch2 cpptrace libassert string_id enkiTS)

# Special handling for Eigen (patching uninstall target)
FetchContent_GetProperties(eigen)
if(NOT eigen_POPULATED)
    FetchContent_Populate(eigen)
    if(EXISTS "${eigen_SOURCE_DIR}/CMakeLists.txt")
        file(READ "${eigen_SOURCE_DIR}/CMakeLists.txt" EIGEN_CMAKE)
        string(REPLACE "add_custom_target ( uninstall" "add_custom_target ( eigen_uninstall" EIGEN_CMAKE "${EIGEN_CMAKE}")
        string(REPLACE "add_custom_target(uninstall" "add_custom_target(eigen_uninstall" EIGEN_CMAKE "${EIGEN_CMAKE}")
        file(WRITE "${eigen_SOURCE_DIR}/CMakeLists.txt" "${EIGEN_CMAKE}")
    endif()
    add_subdirectory(${eigen_SOURCE_DIR} ${eigen_BINARY_DIR})
endif()

if(MSVC)
    set(SCRY_WARNING_FLAGS /W4 /WX)
else()
    set(SCRY_WARNING_FLAGS -Wall -Wextra -Wpedantic -Wno-tautological-compare -Wno-deprecated-enum-enum-conversion -Wno-deprecated-copy -Wno-deprecated-declarations -Wno-template-body -Wno-attributes)
endif()

include(cmake/EngineHelpers.cmake)

# ─── Engine Library ──────────────────────────────────────────────────────────

file(GLOB_RECURSE SCRY_SRC CONFIGURE_DEPENDS "src/*.cpp")
file(GLOB_RECURSE SCRY_INC CONFIGURE_DEPENDS "include/*.hpp" "include/*.h")

add_library(scry SHARED ${SCRY_SRC} ${SCRY_INC})
target_compile_options(scry PRIVATE ${SCRY_WARNING_FLAGS})

# Robust include propagation for dependencies
# We use PUBLIC so both scry and anyone linking against scry (like sandbox and tests) see them.
target_include_directories(scry PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>
    "${CMAKE_CURRENT_SOURCE_DIR}/external/flecs/include"
    "${CMAKE_CURRENT_SOURCE_DIR}/external/mimalloc/include"
    "${CMAKE_CURRENT_SOURCE_DIR}/external/yyjson/src"
    "${CMAKE_CURRENT_SOURCE_DIR}/external/enkiTS/src"
    "${CMAKE_CURRENT_SOURCE_DIR}/external/eigen"
    "${CMAKE_CURRENT_SOURCE_DIR}/external/string_id"
    "${CMAKE_CURRENT_SOURCE_DIR}/external/libassert/include"
    "${CMAKE_CURRENT_SOURCE_DIR}/external/cpptrace/include"
    "${CMAKE_CURRENT_SOURCE_DIR}/external/Catch2/src/catch2"
)

target_compile_definitions(scry PRIVATE ENGINE_EXPORT)

target_link_libraries(scry
    PUBLIC
        mimalloc-static
        flecs
        SDL3::SDL3-static
        yyjson
        libassert::assert
        foonathan_string_id
        enkiTS
)

# ─── Tests ───────────────────────────────────────────────────────────────────

if(SCRY_BUILD_TESTS)
    add_subdirectory(tests)

    # Exclude tests from ALL builds by default
    set_target_properties(scry_tests PROPERTIES EXCLUDE_FROM_ALL TRUE)
    
    # Re-enable for Debug in Multi-Config generators
    if(CMAKE_CONFIGURATION_TYPES)
        foreach(cfg ${CMAKE_CONFIGURATION_TYPES})
            if(cfg STREQUAL "Debug")
                string(TOUPPER ${cfg} cfg_upper)
                # Note: EXCLUDE_FROM_ALL_DEBUG is supported by some generators
                set_target_properties(scry_tests PROPERTIES EXCLUDE_FROM_ALL_${cfg_upper} FALSE)
            endif()
        endforeach()
    endif()

    add_custom_target(run_tests
        COMMAND $<TARGET_FILE:scry_tests> --reporter compact
        DEPENDS scry_tests
        WORKING_DIRECTORY "$<TARGET_FILE_DIR:scry_tests>"
        COMMENT "Running Scry test suite"
        VERBATIM
    )
    
    set_target_properties(run_tests PROPERTIES EXCLUDE_FROM_ALL TRUE)
    if(CMAKE_CONFIGURATION_TYPES)
        foreach(cfg ${CMAKE_CONFIGURATION_TYPES})
            if(cfg STREQUAL "Debug")
                string(TOUPPER ${cfg} cfg_upper)
                set_target_properties(run_tests PROPERTIES EXCLUDE_FROM_ALL_${cfg_upper} FALSE)
            endif()
        endforeach()
    endif()
endif()

# ─── Example sandbox project ──────────────────────────────────────────────────
if(SCRY_BUILD_EXAMPLES)
    add_subdirectory(examples/sandbox)
endif()

cmake_minimum_required(VERSION 3.19)
project(SCRY VERSION 1.0.0 LANGUAGES C CXX)

set(CMAKE_POLICY_VERSION_MINIMUM 3.5)

# Require C++17 and C11
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Configure output directories
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)

# Build options
option(SCRY_BUILD_TESTS "Build unit tests" ON)
option(SCRY_BUILD_EXAMPLES "Build example sandbox" ON)

include(FetchContent)

# ─── Global dependency settings to minimize noise ───
set(BUILD_TESTING OFF CACHE BOOL "" FORCE)
set(CMAKE_SKIP_INSTALL_RULES ON CACHE BOOL "" FORCE)
set(INSTALL_GTEST OFF CACHE BOOL "" FORCE)

# enkiTS
set(ENKITS_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
set(ENKITS_INSTALL OFF CACHE BOOL "" FORCE)

# mimalloc
set(MI_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(MI_BUILD_OBJECT OFF CACHE BOOL "" FORCE)
set(MI_OVERRIDE ON CACHE BOOL "" FORCE)
set(MI_BUILD_SHARED OFF CACHE BOOL "" FORCE)
set(MI_BUILD_STATIC ON CACHE BOOL "" FORCE)

# SDL3
set(SDL_SHARED OFF CACHE BOOL "" FORCE)
set(SDL_STATIC ON CACHE BOOL "" FORCE)
set(SDL_TESTS OFF CACHE BOOL "" FORCE)
set(SDL_INSTALL OFF CACHE BOOL "" FORCE)

# Flecs
set(FLECS_STATIC OFF CACHE BOOL "" FORCE)
set(FLECS_SHARED ON CACHE BOOL "" FORCE)
set(FLECS_TESTS OFF CACHE BOOL "" FORCE)
set(FLECS_EXAMPLES OFF CACHE BOOL "" FORCE)

# Catch2
set(CATCH_BUILD_TESTING OFF CACHE BOOL "" FORCE)
set(CATCH_INSTALL_HELPERS OFF CACHE BOOL "" FORCE)

# foonathan/string_id
set(FOONATHAN_STRING_ID_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
set(FOONATHAN_STRING_ID_BUILD_TESTS OFF CACHE BOOL "" FORCE)

# cpptrace
set(CPPTRACE_INSTALL OFF CACHE BOOL "" FORCE)
set(CPPTRACE_GET_SYMBOLS_WITH_DBGHELP ON CACHE BOOL "" FORCE)
set(CPPTRACE_GET_SYMBOLS_WITH_LIBDWARF OFF CACHE BOOL "" FORCE)

# libassert
set(LIBASSERT_INSTALL OFF CACHE BOOL "" FORCE)
set(LIBASSERT_BUILD_TESTING OFF CACHE BOOL "" FORCE)

)

FetchContent_Declare(Catch2

)

FetchContent_Declare(cpptrace
    GIT_REPOSITORY https://github.com/jeremy-rifkin/cpptrace.git

)

FetchContent_Declare(libassert
    GIT_REPOSITORY https://github.com/jeremy-rifkin/libassert.git

FetchContent_Declare(string_id
    GIT_REPOSITORY https://github.com/foonathan/string_id.git

FetchContent_MakeAvailable(mimalloc SDL3 flecs yyjson Catch2 cpptrace libassert string_id enkiTS eigen)

if(MSVC)
    set(SCRY_WARNING_FLAGS /W4 /WX)
else()
    set(SCRY_WARNING_FLAGS -Wall -Wextra -Wpedantic -Wno-tautological-compare -Wno-deprecated-enum-enum-conversion -Wno-deprecated-copy -Wno-deprecated-declarations -Wno-template-body -Wno-attributes)
endif()

include(cmake/EngineHelpers.cmake)

# ─── Engine Library ──────────────────────────────────────────────────────────

file(GLOB_RECURSE SCRY_SRC CONFIGURE_DEPENDS "src/*.cpp")
file(GLOB_RECURSE SCRY_INC CONFIGURE_DEPENDS "include/*.hpp" "include/*.h")

add_library(scry SHARED ${SCRY_SRC} ${SCRY_INC})
target_compile_options(scry PRIVATE ${SCRY_WARNING_FLAGS})

target_include_directories(scry PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>
    "${eigen_SOURCE_DIR}"
    "${string_id_SOURCE_DIR}"
)

target_compile_definitions(scry PRIVATE ENGINE_EXPORT)

target_link_libraries(scry
    PUBLIC
        mimalloc-static
        flecs
        SDL3::SDL3-static
        yyjson
        libassert::assert
        foonathan_string_id
        enkiTS
)

# ─── Tests ───────────────────────────────────────────────────────────────────

if(SCRY_BUILD_TESTS)
    add_subdirectory(tests)

    # Exclude tests from ALL builds by default
    set_target_properties(scry_tests PROPERTIES EXCLUDE_FROM_ALL TRUE)
    
    # Re-enable for Debug in Multi-Config generators
    if(CMAKE_CONFIGURATION_TYPES)
        foreach(cfg ${CMAKE_CONFIGURATION_TYPES})
            if(cfg STREQUAL "Debug")
                string(TOUPPER ${cfg} cfg_upper)
                set_target_properties(scry_tests PROPERTIES EXCLUDE_FROM_ALL_${cfg_upper} FALSE)
            endif()
        endforeach()
    endif()

    add_custom_target(run_tests
        COMMAND $<TARGET_FILE:scry_tests> --reporter compact
        DEPENDS scry_tests
        WORKING_DIRECTORY "$<TARGET_FILE_DIR:scry_tests>"
        COMMENT "Running Scry test suite"
        VERBATIM
    )
    
    set_target_properties(run_tests PROPERTIES EXCLUDE_FROM_ALL TRUE)
    if(CMAKE_CONFIGURATION_TYPES)
        foreach(cfg ${CMAKE_CONFIGURATION_TYPES})
            if(cfg STREQUAL "Debug")
                string(TOUPPER ${cfg} cfg_upper)
                set_target_properties(run_tests PROPERTIES EXCLUDE_FROM_ALL_${cfg_upper} FALSE)
            endif()
        endforeach()
    endif()
endif()

if(SCRY_BUILD_EXAMPLES)
    add_subdirectory(examples/sandbox)

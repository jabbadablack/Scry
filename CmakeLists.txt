# Define macro for plugin building
macro(scry_add_plugin target_name)
    if(SCRY_IS_PLUGIN)
        file(GLOB_RECURSE PLUGIN_SRC CONFIGURE_DEPENDS "*.cpp" "*.c")
        add_library(${target_name} SHARED ${PLUGIN_SRC})
        target_link_libraries(${target_name} PRIVATE scry flecs)
        target_include_directories(${target_name} PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}/../../include")
        
        # Copy plugin to the binary output directory next to scry DLL
        add_custom_command(TARGET ${target_name} POST_BUILD
            COMMAND ${CMAKE_COMMAND} -E make_directory $<TARGET_FILE_DIR:scry>/plugins
            COMMAND ${CMAKE_COMMAND} -E copy_if_different
            $<TARGET_FILE:${target_name}>
            $<TARGET_FILE_DIR:scry>/plugins/$<TARGET_FILE_NAME:${target_name}>
        )
    else()
        message(FATAL_ERROR "Plugin ${target_name} cannot be built: SCRY_IS_PLUGIN is not set to true")
    endif()
endmacro()

# JSON Dependency Injection for Plugins
if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/project.json")
    file(READ "${CMAKE_CURRENT_SOURCE_DIR}/project.json" PROJECT_JSON)
    string(JSON PLUGINS_LEN LENGTH "${PROJECT_JSON}" "plugins")
    math(EXPR loop_end "${PLUGINS_LEN} - 1")
    if(loop_end GREATER_EQUAL 0)
        foreach(idx RANGE ${loop_end})
            string(JSON plugin_name GET "${PROJECT_JSON}" "plugins" ${idx})
            set(plugin_json_path "${CMAKE_CURRENT_SOURCE_DIR}/plugins/${plugin_name}/plugin.json")
            if(EXISTS "${plugin_json_path}")
                file(READ "${plugin_json_path}" PLUGIN_JSON)
                string(JSON parsed_name GET "${PLUGIN_JSON}" "name")
                if(parsed_name STREQUAL plugin_name)
                    set(SCRY_IS_PLUGIN TRUE)
                    add_subdirectory("plugins/${plugin_name}")
                else()
                    message(WARNING "Plugin name mismatch for ${plugin_name}: got ${parsed_name}")
                endif()
            else()
                message(WARNING "plugin.json not found for ${plugin_name} at ${plugin_json_path}")
            endif()
        endforeach()
    endif()
endif()

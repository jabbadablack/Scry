cmake_minimum_required(VERSION 3.18)
project(SCRY VERSION 1.0.0 LANGUAGES C CXX)

# Require C++20 and C11
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Configure mimalloc options
set(MI_OVERRIDE ON CACHE BOOL "Override standard malloc" FORCE)
set(MI_BUILD_SHARED OFF CACHE BOOL "Build mimalloc as a shared library" FORCE)
set(MI_BUILD_OBJECT OFF CACHE BOOL "Build mimalloc object files" FORCE)
set(MI_BUILD_TESTS OFF CACHE BOOL "Build mimalloc tests" FORCE)
set(MI_BUILD_STATIC ON CACHE BOOL "Build mimalloc as static library" FORCE)

# Configure SDL3 options to minimize compile time
set(SDL_SHARED OFF CACHE BOOL "Build shared SDL library" FORCE)
set(SDL_STATIC ON CACHE BOOL "Build static SDL library" FORCE)
set(SDL_TESTS OFF CACHE BOOL "Build SDL tests" FORCE)
set(SDL_INSTALL OFF CACHE BOOL "Install SDL" FORCE)
set(SDL_AUDIO OFF CACHE BOOL "Disable SDL Audio subsystem" FORCE)
set(SDL_JOYSTICK OFF CACHE BOOL "Disable SDL Joystick subsystem" FORCE)
set(SDL_HAPTIC OFF CACHE BOOL "Disable SDL Haptic subsystem" FORCE)
set(SDL_SENSOR OFF CACHE BOOL "Disable SDL Sensor subsystem" FORCE)
set(SDL_CAMERA OFF CACHE BOOL "Disable SDL Camera subsystem" FORCE)

# Configure Flecs options
set(FLECS_SHARED ON CACHE BOOL "Build Flecs shared library" FORCE)
set(FLECS_STATIC OFF CACHE BOOL "Build Flecs static library" FORCE)
set(FLECS_TESTS OFF CACHE BOOL "Build Flecs tests" FORCE)

# Integrate mimalloc, SDL3, & Flecs via FetchContent
include(FetchContent)

FetchContent_Declare(
    mimalloc
    GIT_REPOSITORY https://github.com/microsoft/mimalloc.git
    GIT_TAG v2.1.2
)

FetchContent_Declare(
    SDL3
    GIT_REPOSITORY https://github.com/libsdl-org/SDL.git
    GIT_TAG release-3.2.0
)

FetchContent_Declare(
    flecs
    GIT_REPOSITORY https://github.com/SanderMertens/flecs.git
    GIT_TAG v4.1.5
)

FetchContent_MakeAvailable(mimalloc SDL3 flecs)

# Enable all pedantic compiler warnings and treat warnings as errors
if(CMAKE_CXX_COMPILER_ID MATCHES "GNU" OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
    add_compile_options(-Wall -Wextra -Wpedantic -Werror)
elseif(MSVC)
    add_compile_options(/W4 /WX)
endif()

# Scry Shared Library
add_library(scry SHARED
    src/scry_memory.cpp
    src/scry_platform.cpp
    src/scry_ecs.cpp
    src/scry_plugin.cpp
    src/scry.cpp
)

# Set Scry include directories
target_include_directories(scry PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
    $<INSTALL_INTERFACE:include>
)

# Export macro definitions
target_compile_definitions(scry PRIVATE SCRY_EXPORT)

# Link dependencies to the shared library scry
target_link_libraries(scry PRIVATE mimalloc-static SDL3-static flecs)
target_include_directories(scry PRIVATE "${mimalloc_SOURCE_DIR}/include")

# Sandbox executable
add_executable(sandbox
    sandbox/main.cpp
)

# Link sandbox against scry, flecs, and mimalloc-static
target_link_libraries(sandbox PRIVATE scry flecs mimalloc-static)
target_include_directories(sandbox PRIVATE "${mimalloc_SOURCE_DIR}/include")


# Copy flecs DLL to sandbox output directory for runtime execution
add_custom_command(TARGET sandbox POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    $<TARGET_FILE:flecs>
    $<TARGET_FILE_DIR:sandbox>
)

# Scry Physics Plugin (Decoupled DLL)
add_library(scry_physics_plugin SHARED
    plugins/scry_physics_plugin/scry_physics_plugin.cpp
)
target_link_libraries(scry_physics_plugin PRIVATE flecs)
target_include_directories(scry_physics_plugin PRIVATE "${CMAKE_CURRENT_SOURCE_DIR}/include")

# Copy the physics plugin DLL to plugins/ directory next to sandbox
add_custom_command(TARGET scry_physics_plugin POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E make_directory $<TARGET_FILE_DIR:sandbox>/plugins
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
    $<TARGET_FILE:scry_physics_plugin>
    $<TARGET_FILE_DIR:sandbox>/plugins/$<TARGET_FILE_NAME:scry_physics_plugin>
)
